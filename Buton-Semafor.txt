<body>
    <div class="container text-center mt-5">
        <button id="trafficButton" class="btn btn-red">Apasă-mă</button>
    </div>

    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const button = document.getElementById('trafficButton');
            const colors = ['btn-red', 'btn-yellow', 'btn-green'];
            let currentColorIndex = 0;

            const changeColor = () => {
                button.classList.remove(colors[currentColorIndex]);
                currentColorIndex = (currentColorIndex + 1) % colors.length;
                button.classList.add(colors[currentColorIndex]);
            };

            button.addEventListener('click', changeColor);

            setInterval(changeColor, 10000); // Schimbă culoarea din 10 în 10 secunde
        });
    </script>
